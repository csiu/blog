## Obtaining tweets with the emoji of interest

```r
get_tweet_with_emoji <- function(emoji){
  popular_emoji_df <-
    emoji_counts %>%

    ## get tweet id containing the emoij
    filter(
      native == emoji,
      counts > 0
    ) %>%
    {.$tweet_id} %>%

    ## get tweet df containing the emoji
    {df_tidy[.,]}

  ## extract text
  (popular_emoji_text <-
    df %>%
    filter(id %in% popular_emoji_df$id) %>%
    {.$text}
  )
}
```
